import { Playground } from 'docz';
import unifiedDoc from 'unified-doc';

# Attributes

When a `doc` is initialized with `filename` and `content` as required fields, we can access these passthrough fields easily in the following example.

<Playground>
  {() => {
    /** ====== START EDIT ======  **/
    const content = '> **some** markdown content';
    const filename = 'doc.md';
    /** ====== END EDIT ======  **/
    const doc = unifiedDoc({ content, filename });
    return (
      <div>
        <h4>Content</h4>
        <pre>{doc.content}</pre>
        <h4>Filename</h4>
        <pre>{doc.filename}</pre>
      </div>
    );
  }}
</Playground>
